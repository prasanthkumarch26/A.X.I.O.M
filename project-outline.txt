async def generate_response(paper_id, mode, question=None):
    cache_key = hash(paper_id, mode, question)

    # 1. Cache hit
    cached = await redis.get(cache_key)
    if cached:
        return cached

    # 2. Fetch paper
    paper = await get_paper(paper_id)

    # 3. Build prompt
    prompt = build_prompt(paper, mode, question)

    try:
        # 4. Fail-fast timeout
        response = await asyncio.wait_for(
            call_ollama(prompt),
            timeout=1.5
        )
    except asyncio.TimeoutError:
        return {"error": "LLM timeout"}

    # 5. Cache result
    await redis.set(cache_key, response, ex=3600)

    return response
